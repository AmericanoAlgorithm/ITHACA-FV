61 void reducedSimpleSteadyNS::solveOnline_Simple(scalar mu_now,
62  scalar NmodesUproj, scalar NmodesPproj)
63 {
64  counter++;
65  scalar UprojN;
66  scalar PprojN;
67 
68  if (NmodesUproj == 0)
69  {
70  UprojN = ULmodes.size();
71  }
72  else
73  {
74  UprojN = NmodesUproj;
75  }
76 
77  if (NmodesPproj == 0)
78  {
79  PprojN = problem->Pmodes.size();
80  }
81  else
82  {
83  PprojN = NmodesPproj;
84  }
85 
86  Eigen::VectorXd uresidualOld = Eigen::VectorXd::Zero(UprojN);
87  Eigen::VectorXd presidualOld = Eigen::VectorXd::Zero(UprojN);
88  Eigen::VectorXd uresidual;
89  Eigen::VectorXd presidual;
90  scalar U_norm_res(1);
91  scalar P_norm_res(1);
92  Eigen::MatrixXd a = Eigen::VectorXd::Zero(UprojN);
93  Eigen::MatrixXd b = Eigen::VectorXd::Zero(PprojN);
94  ITHACAparameters para;
95  float residualJumpLim =
96  para.ITHACAdict->lookupOrDefault<float>("residualJumpLim", 1e-5);
97  float normalizedResidualLim =
98  para.ITHACAdict->lookupOrDefault<float>("normalizedResidualLim", 1e-5);
99  scalar residual_jump(1 + residualJumpLim);
100  volVectorField Uaux("Uaux", problem->_U());
101  volScalarField Paux("Paux", problem->_p());
102  int iter = 0;
103 
104  while (residual_jump > residualJumpLim
105  || std::max(U_norm_res, P_norm_res) > normalizedResidualLim)
106  {
107  iter++;
108  Uaux = ULmodes.reconstruct(a, "Uaux");
109  Paux = problem->Pmodes.reconstruct(b, "Paux");
110  simpleControl& simple = problem->_simple();
111  setRefCell(Paux, simple.dict(), problem->pRefCell, problem->pRefValue);
112  problem->_phi() = linearInterpolate(Uaux) & problem->_U().mesh().Sf();
113  fvVectorMatrix Au(get_Umatrix_Online(Uaux, Paux));
114  List<Eigen::MatrixXd> RedLinSysU = ULmodes.project(Au, UprojN);
115  a = reducedProblem::solveLinearSys(RedLinSysU, a, uresidual, vel_now);
116  //Info << uresidual.norm() << endl;
117  Uaux = ULmodes.reconstruct(a, "Uaux");
118  problem->_phi() = linearInterpolate(Uaux) & problem->_U().mesh().Sf();
119  fvScalarMatrix Ap(get_Pmatrix_Online(Uaux, Paux));
120  List<Eigen::MatrixXd> RedLinSysP = problem->Pmodes.project(Ap, PprojN);
121  b = reducedProblem::solveLinearSys(RedLinSysP, b, presidual);
122  //Info << presidual.norm() << endl;
123  uresidualOld = uresidualOld - uresidual;
124  presidualOld = presidualOld - presidual;
125  uresidualOld = uresidualOld.cwiseAbs();
126  presidualOld = presidualOld.cwiseAbs();
127  //std::cout << uresidualOld.sum() << std::endl;
128  //std::cout << presidualOld.sum() << std::endl;
129  residual_jump = std::max(uresidualOld.sum(), presidualOld.sum());
130  uresidualOld = uresidual;
131  presidualOld = presidual;
132  uresidual = uresidual.cwiseAbs();
133  presidual = presidual.cwiseAbs();
134  U_norm_res = uresidual.sum() / (RedLinSysU[1].cwiseAbs()).sum();
135  P_norm_res = presidual.sum() / (RedLinSysP[1].cwiseAbs()).sum();
136  // std::cout << "Residual jump = " << residual_jump << std::endl;
137  // std::cout << "Normalized residual = " << std::max(U_norm_res,P_norm_res) << std::endl;
138  }
139 
140  std::cout << "Solution " << counter << " converged in " << iter <<
141  " iterations." << std::endl;
142  std::cout << "Final normalized residual for velocity: " << U_norm_res <<
143  std::endl;
144  std::cout << "Final normalized residual for pressure: " << P_norm_res <<
145  std::endl;
146  Uaux = ULmodes.reconstruct(a, "Uaux");
147  Paux = problem->Pmodes.reconstruct(b, "Paux");
148  ITHACAstream::exportSolution(Uaux, name(counter),
149  "./ITHACAoutput/Reconstruct/");
150  ITHACAstream::exportSolution(Paux, name(counter),
151  "./ITHACAoutput/Reconstruct/");
152 }

154 fvVectorMatrix reducedSimpleSteadyNS::get_Umatrix_Online(volVectorField& U,
155  volScalarField& p)
156 {
157  IOMRFZoneList& MRF = problem->_MRF();
158  surfaceScalarField& phi = problem->_phi();
159  fv::options& fvOptions = problem->_fvOptions();
160  MRF.correctBoundaryVelocity(U);
161  fvVectorMatrix Ueqn
162  (
163  fvm::div(phi, U)
164  + MRF.DDt(U)
165  + problem->turbulence->divDevReff(U)
166  ==
167  fvOptions(U)
168  );
169  problem->Ueqn_global = &Ueqn;
170  return Ueqn;
171 }

173 fvScalarMatrix reducedSimpleSteadyNS::get_Pmatrix_Online(volVectorField& U,
174  volScalarField& p)
175 {
176  IOMRFZoneList& MRF = problem->_MRF();
177  MRF.correctBoundaryVelocity(U);
178  volScalarField rAU(1.0 / problem->Ueqn_global->A());
179  volVectorField HbyA(constrainHbyA(rAU * problem->Ueqn_global->H(), U, p));
180  surfaceScalarField phiHbyA("phiHbyA", fvc::flux(HbyA));
181  MRF.makeRelative(phiHbyA);
182  adjustPhi(phiHbyA, U, p);
183  tmp<volScalarField> rAtU(rAU);
184  fvScalarMatrix pEqn
185  (
186  fvm::laplacian(rAtU(), p) == fvc::div(phiHbyA)
187  );
188  return pEqn;
189 }
