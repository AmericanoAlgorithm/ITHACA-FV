90 //Operator to evaluate the residual for the supremizer approach
91 int newton_unsteadyBB_sup::operator()(const Eigen::VectorXd& x,
92  Eigen::VectorXd& fvec) const
93 {
94  Eigen::VectorXd a_dot(Nphi_u);
95  Eigen::VectorXd a_tmp(Nphi_u);
96  Eigen::VectorXd b_tmp(Nphi_prgh);
97  Eigen::VectorXd c_dot(Nphi_t);
98  Eigen::VectorXd c_tmp(Nphi_t);
99  a_tmp = x.head(Nphi_u);
100  a_dot = (x.head(Nphi_u) - y_old.head(Nphi_u)) / dt;
101  b_tmp = x.segment(Nphi_u, Nphi_prgh);
102  c_tmp = x.tail(Nphi_t);
103  c_dot = (x.tail(Nphi_t) - y_old.tail(Nphi_t)) / dt;
104  // Convective term
105  Eigen::MatrixXd cc(1, 1);
106  // Diffusive Term
107  Eigen::VectorXd M1 = problem->B_matrix * a_tmp * nu;
108  // Mass Term Velocity
109  Eigen::VectorXd M5 = problem->M_matrix * a_dot;
110  // Gradient of pressure
111  Eigen::VectorXd M2 = problem->K_matrix * b_tmp;
112  // Continuity
113  Eigen::VectorXd M3 = problem->P_matrix * a_tmp;
114  // Buoyancy Term
115  Eigen::VectorXd M10 = problem->H_matrix * c_tmp;
116  // Convective term temperature
117  Eigen::MatrixXd qq(1, 1);
118  // diffusive term temperature
119  Eigen::VectorXd M6 = problem->Y_matrix * c_tmp * (nu / Pr);
120  // Mass Term Temperature
121  Eigen::VectorXd M8 = problem->W_matrix * c_dot;
122 
123  for (label i = 0; i < Nphi_u; i++)
124  {
125  cc = a_tmp.transpose() * Eigen::SliceFromTensor(problem->C_tensor, 0,
126  i) * a_tmp;
127  fvec(i) = - M5(i) + M1(i) - cc(0, 0) - M10(i) - M2(i);
128  }
129 
130  for (label j = 0; j < Nphi_prgh; j++)
131  {
132  label k = j + Nphi_u;
133  fvec(k) = M3(j);
134  }
135 
136  for (label j = 0; j < Nphi_t; j++)
137  {
138  label k = j + Nphi_u + Nphi_prgh;
139  qq = a_tmp.transpose() * problem->Q_matrix[j] * c_tmp;
140  fvec(k) = -M8(j) + M6(j) - qq(0, 0);
141  }
142 
143  for (label j = 0; j < N_BC; j++)
144  {
145  fvec(j) = x(j) - BC(j);
146  }
147 
148  for (label j = 0; j < N_BC_t; j++)
149  {
150  label k = j + Nphi_u + Nphi_prgh;
151  fvec(k) = x(k) - BC_t(j);
152  }
153 
154  return 0;
155 }

168 Eigen::MatrixXd reducedUnsteadyBB::solveOnline_sup(Eigen::MatrixXd& temp_now_BC,
169  Eigen::MatrixXd& vel_now_BC, label NParaSet, label startSnap)
170 {
171  std::cout << "################## Online solve N° " << NParaSet <<
172  " ##################" << std::endl;
173  std::cout << "Solving for the parameter: " << temp_now_BC << std::endl;
174  // Count number of time steps
175  int counter = 0;
176  time = tstart;
177 
178  while (time < finalTime - 0.5 * dt)
179  {
180  time = time + dt;
181  counter ++;
182  }
183 
184  // Set size of online solution
185  online_solutiont.resize(Nphi_u + Nphi_prgh + Nphi_t + 1, counter + 1);
186  // Set initial condition for online solve
187  volScalarField T_IC("T_IC", problem->Tfield[0]);
188 
189  for (label j = 0; j < T_IC.boundaryField().size(); j++)
190  {
191  for (label i = 0; i < N_BC_t; i++)
192  {
193  if (j == problem->inletIndexT(i, 0))
194  {
195  T_IC.boundaryFieldRef()[problem->inletIndexT(i, 0)][j] = temp_now_BC(i, 0);
196  }
197  else
198  {
199  }
200  }
201  }
202 
203  // Create and resize the solution vector
204  y.resize(Nphi_u + Nphi_prgh + Nphi_t, 1);
205  y.setZero();
206  // Calculate the time-dependent coefficients
207  y.head(Nphi_u) = ITHACAutilities::get_coeffs(problem->Ufield[startSnap],
208  LUmodes);
209 
210  if (Nphi_prgh != 0)
211  {
212  y.segment(Nphi_u, Nphi_prgh) = ITHACAutilities::get_coeffs_ortho(
213  problem->Prghfield[2],
214  problem->Prghmodes);
215  }
216 
217  y.tail(Nphi_t) = ITHACAutilities::get_coeffs(T_IC, LTmodes);
218  // Set some properties of the newton object
219  newton_object_sup.nu = nu;
220  newton_object_sup.y_old = y;
221  newton_object_sup.dt = dt;
222  newton_object_sup.Pr = Pr;
223  newton_object_sup.BC_t.resize(N_BC_t);
224  newton_object_sup.BC.resize(N_BC);
225 
226  // Change initial condition for the lifting function
227  for (label j = 0; j < N_BC_t; j++)
228  {
229  newton_object_sup.BC_t(j) = temp_now_BC(j, 0);
230  }
231 
232  for (label j = 0; j < N_BC; j++)
233  {
234  newton_object_sup.BC(j) = vel_now_BC(j, 0);
235  }
236 
237  // Set the initial time
238  time = tstart;
239  // Create vector to store temporal solution and save initial condition as first solution
240  Eigen::MatrixXd tmp_sol(Nphi_u + Nphi_prgh + Nphi_t + 1, 1);
241  tmp_sol(0) = time;
242  tmp_sol.col(0).tail(y.rows()) = y;
243  online_solutiont.col(0) = tmp_sol;
244  // Create nonlinear solver object
245  Eigen::HybridNonLinearSolver<newton_unsteadyBB_sup> hnls(newton_object_sup);
246  // Set output colors for fancy output
247  Color::Modifier red(Color::FG_RED);
248  Color::Modifier green(Color::FG_GREEN);
249  Color::Modifier def(Color::FG_DEFAULT);
250 
251  // Start the time loop
252  for (label i = 1; i < online_solutiont.cols(); i++)
253  {
254  time = time + dt;
255  Eigen::VectorXd res(y);
256  res.setZero();
257  hnls.solve(y);
258 
259  for (label j = 0; j < N_BC; j++)
260  {
261  y(j) = vel_now_BC(j, 0);
262  }
263 
264  for (label j = 0; j < N_BC_t; j++)
265  {
266  label k = j + Nphi_prgh + Nphi_u;
267  y(k) = temp_now_BC(j, 0);
268  }
269 
270  newton_object_sup.operator()(y, res);
271  newton_object_sup.y_old = y;
272  Info << "Time = " << time << endl;
273 
274  if (res.norm() < 1e-5)
275  {
276  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
277  hnls.iter << " iterations " << def << std::endl << std::endl;
278  }
279  else
280  {
281  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
282  hnls.iter << " iterations " << def << std::endl << std::endl;
283  }
284 
285  tmp_sol(0) = time;
286  tmp_sol.col(0).tail(y.rows()) = y;
287  online_solutiont.col(i) = tmp_sol;
288  }
289 
290  // Save the current solution
291  ITHACAstream::exportMatrix(online_solutiont, "red_coeff", "python",
292  "./ITHACAoutput/red_coeff/" + name(NParaSet) + "/");
293  ITHACAstream::exportMatrix(online_solutiont, "red_coeff", "matlab",
294  "./ITHACAoutput/red_coeff/" + name(NParaSet) + "/");
295  ITHACAstream::exportMatrix(online_solutiont, "red_coeff", "eigen",
296  "./ITHACAoutput/red_coeff/" + name(NParaSet) + "/");
297  return online_solutiont;
298 }

302 void reducedUnsteadyBB::reconstruct_sup(fileName folder, int printevery)
303 {
304  if (ITHACAutilities::check_folder(folder))
305  {
306  }
307  else
308  {
309  mkDir(folder);
310  ITHACAutilities::createSymLink(folder);
311  }
312 
313  int counter = 0;
314  int nextwrite = 0;
315  int counter2 = 1 + TREC.size();
316 
317  for (label i = 0; i < online_solutiont.cols(); i++)
318  {
319  if (counter == nextwrite)
320  {
321  volVectorField U_rec("U_rec", LUmodes[0] * 0);
322 
323  for (label j = 0; j < Nphi_u; j++)
324  {
325  U_rec += LUmodes[j] * online_solutiont(j + 1, i);
326  }
327 
328  ITHACAstream::exportSolution(U_rec, name(counter2), folder);
329 
330  if (Nphi_prgh != 0)
331  {
332  volScalarField P_rec("P_rec", problem->Prghmodes[0] * 0);
333 
334  for (label j = 0; j < Nphi_prgh; j++)
335  {
336  P_rec += problem->Prghmodes[j] * online_solutiont(j + Nphi_u + 1, i);
337  }
338 
339  ITHACAstream::exportSolution(P_rec, name(counter2), folder);
340  PREC.append(P_rec);
341  }
342 
343  volScalarField T_rec("T_rec", LTmodes[0] * 0);
344 
345  for (label j = 0; j < Nphi_t; j++)
346  {
347  T_rec += LTmodes[j] * online_solutiont(j + Nphi_prgh + Nphi_u + 1, i);
348  }
349 
350  ITHACAstream::exportSolution(T_rec, name(counter2), folder);
351  nextwrite += printevery;
352  double timenow = online_solutiont(0, i);
353  std::ofstream of(folder + name(counter2) + "/" + name(timenow));
354  counter2 ++;
355  UREC.append(U_rec);
356  TREC.append(T_rec);
357  }
358 
359  counter++;
360  }
