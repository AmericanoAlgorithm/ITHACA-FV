44 class DEIM_function : public DEIM<PtrList<volScalarField>, volScalarField >
45 {
46 public:
47  using DEIM::DEIM;
48  static volScalarField evaluate_expression(volScalarField& S, Eigen::MatrixXd mu)
49  {
50  volScalarField yPos = S.mesh().C().component(vector::Y);
51  volScalarField xPos = S.mesh().C().component(vector::X);
52  for (auto i = 0; i < S.size(); i++)
53  {
54  S[i] = std::exp( - 2 * std::pow(xPos[i] - mu(0) - 1, 2) - 2 * std::pow(yPos[i] - mu(1) - 0.5 , 2));
55  }
56  return S;
57  }
58  Eigen::VectorXd onlineCoeffs(Eigen::MatrixXd mu)
59  {
60  theta.resize(fields.size());
61  for (int i = 0; i < fields.size(); i++)
62  {
63  double on_coeff = evaluate_expression(fields[i], mu)[localMagicPoints[i]];
64  theta(i) = on_coeff;
65  }
66  return theta;
67  }
68 };
