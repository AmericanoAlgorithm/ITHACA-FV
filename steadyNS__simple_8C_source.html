71 fvScalarMatrix steadyNS_simple::get_Pmatrix(volVectorField& U,
72  volScalarField& p, scalar& presidual)
73 {
74  IOMRFZoneList& MRF = _MRF();
75  surfaceScalarField& phi = _phi();
76  simpleControl& simple = _simple();
77  fv::options& fvOptions = _fvOptions();
78  MRF.correctBoundaryVelocity(U);
79  fvMesh& mesh = _mesh();
80  volScalarField rAU(1.0 / Ueqn_global->A());
81  volVectorField HbyA(constrainHbyA(rAU * Ueqn_global->H(), U, p));
82  surfaceScalarField phiHbyA("phiHbyA", fvc::flux(HbyA));
83  MRF.makeRelative(phiHbyA);
84  adjustPhi(phiHbyA, U, p);
85  tmp<volScalarField> rAtU(rAU);
86 
87  if (simple.consistent())
88  {
89  rAtU = 1.0 / (1.0 / rAU - Ueqn_global->H1());
90  phiHbyA +=
91  fvc::interpolate(rAtU() - rAU) * fvc::snGrad(p) * mesh.magSf();
92  HbyA -= (rAU - rAtU()) * fvc::grad(p);
93  }
94 
95  constrainPressure(p, U, phiHbyA, rAtU(), MRF);
96  int i = 0;
97 
98  while (simple.correctNonOrthogonal())
99  {
100  fvScalarMatrix pEqn
101  (
102  fvm::laplacian(rAtU(), p) == fvc::div(phiHbyA)
103  );
104  pEqn.setReference(pRefCell, pRefValue);
105 
106  if (i == 0)
107  {
108  presidual = pEqn.solve().initialResidual();
109  }
110  else
111  {
112  pEqn.solve().initialResidual();
113  }
114 
115  if (simple.finalNonOrthogonalIter())
116  {
117  phi = phiHbyA - pEqn.flux();
118  }
119 
120  i++;
121  }
122 
123  //p.storePrevIter(); // Perché ho dovuto metterlo se nel solver non c'è???
124  p.relax();
125  U = HbyA - rAtU() * fvc::grad(p);
126  U.correctBoundaryConditions();
127  fvOptions.correct(U);
128  fvScalarMatrix pEqn
129  (
130  fvm::laplacian(rAtU(), p) == fvc::div(phiHbyA)
131  );
132  return pEqn;
133 }
