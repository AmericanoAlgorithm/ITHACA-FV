597 template<typename T>
598 Eigen::MatrixXd ITHACAutilities::get_mass_matrix_Eigen(PtrList<T>& fields)
599 {
600  Eigen::MatrixXd F = Foam2Eigen::PtrList2Eigen(fields);
601  Eigen::VectorXd V = Foam2Eigen::field2Eigen(fields[0].mesh().V());
602  Eigen::MatrixXd VM = V.asDiagonal();
603  Eigen::MatrixXd M = F.transpose() * VM * F;
604  return M;
605 }
