503 template<typename T>
504 Eigen::MatrixXd ITHACAutilities::get_mass_matrix_Eigen(PtrList<T>& fields)
505 {
506  Eigen::MatrixXd F = Foam2Eigen::PtrList2Eigen(fields);
507  Eigen::VectorXd V = Foam2Eigen::field2Eigen(fields[0].mesh().V());
508  Eigen::MatrixXd VM = V.asDiagonal();
509  Eigen::MatrixXd M = F.transpose() * VM * F;
510  return M;
511 }
