740 template<typename T>
741 Eigen::MatrixXd ITHACAutilities::get_mass_matrix_Eigen(PtrList<T>& fields,
742  bool consider_volumes)
743 {
744  Eigen::MatrixXd F = Foam2Eigen::PtrList2Eigen(fields);
745  Eigen::VectorXd V = Foam2Eigen::field2Eigen(fields[0].mesh().V());
746  Eigen::MatrixXd VM = V.asDiagonal();
747  Eigen::MatrixXd M;
748 
749  if (consider_volumes)
750  {
751  M = F.transpose() * VM * F;
752  }
753  else
754  {
755  M = F.transpose() * F;
756  }
757 
758  return M;
759 }
