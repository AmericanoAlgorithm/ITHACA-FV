432 template <typename T, int whatever, typename IND>
433 void ITHACAstream::SaveSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m,
434  word folder, word MatrixName)
435 {
436  typedef Eigen::Triplet<int> Trip;
437  std::vector<Trip> res;
438  m.makeCompressed();
439  mkDir(folder);
440  std::fstream writeFile;
441  writeFile.open(folder + MatrixName, std::ios::binary | std::ios::out);
442 
443  if (writeFile.is_open())
444  {
445  IND rows, cols, nnzs, outS, innS;
446  rows = m.rows() ;
447  cols = m.cols() ;
448  nnzs = m.nonZeros() ;
449  outS = m.outerSize();
450  innS = m.innerSize();
451  writeFile.write(reinterpret_cast<const char*> (&rows), sizeof(IND));
452  writeFile.write(reinterpret_cast<const char*> (&cols), sizeof(IND));
453  writeFile.write(reinterpret_cast<const char*> (&nnzs), sizeof(IND));
454  writeFile.write(reinterpret_cast<const char*> (&outS), sizeof(IND));
455  writeFile.write(reinterpret_cast<const char*> (&innS), sizeof(IND));
456  writeFile.write(reinterpret_cast<const char*>(m.valuePtr()),
457  sizeof(T ) * m.nonZeros());
458  writeFile.write(reinterpret_cast<const char*>(m.outerIndexPtr()),
459  sizeof(IND) * m.outerSize());
460  writeFile.write(reinterpret_cast<const char*>(m.innerIndexPtr()),
461  sizeof(IND) * m.nonZeros());
462  writeFile.close();
463  }

466 template <typename T, int whatever, typename IND>
467 void ITHACAstream::ReadSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m,
468  word folder, word MatrixName)
469 {
470  std::fstream readFile;
471  readFile.open(folder + MatrixName, std::ios::binary | std::ios::in);
472 
473  if (!readFile.good())
474  {
475  std::cout << folder + MatrixName <<
476  " file does not exist, try to rerun the Offline Stage!" << std::endl;
477  exit(EXIT_FAILURE);
478  }
479  else if (readFile.is_open())
480  {
481  IND rows, cols, nnz, inSz, outSz;
482  readFile.read(reinterpret_cast<char*>(&rows ), sizeof(IND));
483  readFile.read(reinterpret_cast<char*>(&cols ), sizeof(IND));
484  readFile.read(reinterpret_cast<char*>(&nnz ), sizeof(IND));
485  readFile.read(reinterpret_cast<char*>(&inSz ), sizeof(IND));
486  readFile.read(reinterpret_cast<char*>(&outSz), sizeof(IND));
487  m.resize(rows, cols);
488  m.makeCompressed();
489  m.resizeNonZeros(nnz);
490  readFile.read(reinterpret_cast<char*>(m.valuePtr()), sizeof(T ) * nnz );
491  readFile.read(reinterpret_cast<char*>(m.outerIndexPtr()), sizeof(IND) * outSz);
492  readFile.read(reinterpret_cast<char*>(m.innerIndexPtr()), sizeof(IND) * nnz );
493  m.finalize();
494  readFile.close();
495  }
496 }
