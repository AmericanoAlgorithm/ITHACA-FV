60 template<class T>
61 List<Eigen::MatrixXd> Modes<T>::project(fvMatrix<T>& Af)
62 {
63  List<Eigen::MatrixXd> LinSys;
64  LinSys.resize(2);
65 
66  if (EigenModes.size() == 0)
67  {
68  toEigen();
69  }
70 
71  Eigen::SparseMatrix<double> Ae;
72  Eigen::VectorXd be;
73  Foam2Eigen::fvMatrix2Eigen(Af, Ae, be);
74  LinSys[0] = EigenModes[0].transpose() * Ae * EigenModes[0];
75  LinSys[1] = EigenModes[0].transpose() * be;
76  return LinSys;
