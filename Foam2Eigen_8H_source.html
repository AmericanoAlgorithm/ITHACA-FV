284 template <class type_m, class type_PtrList>
285 std::tuple<Eigen::MatrixXd, Eigen::VectorXd> Foam2Eigen::projectFvMatrix(
286  type_m& matrix, type_PtrList& modes, int Nmodes)
287 {
288  Eigen::SparseMatrix<double> A;
289  Eigen::MatrixXd Ar;
290  Eigen::VectorXd b;
291  Eigen::VectorXd br;
292  Eigen::MatrixXd Eig_Modes = PtrList2Eigen(modes, Nmodes);
293  Foam2Eigen::fvMatrix2Eigen(matrix, A, b);
294  Eigen::VectorXd Volumes = field2Eigen(modes[0].mesh());
295  Eigen::MatrixXd VolumesN(Volumes.rows(), Nmodes);
296 
297  if (Volumes.rows() != Eig_Modes.rows())
298  {
299  VolumesN.resize(Eig_Modes.rows(), Nmodes);
300  }
301 
302  if (Volumes.rows() == Eig_Modes.rows())
303  {
304  for (auto i = 0; i < Nmodes; i++)
305  {
306  VolumesN.col(i) = Volumes;
307  }
308  }
309  else
310  {
311  for (auto i = 0; i < Nmodes; i++)
312  {
313  VolumesN.col(i).segment(0, Volumes.rows()) = Volumes;
314  VolumesN.col(i).segment(Volumes.rows() + 1, Volumes.rows() * 2) = Volumes;
315  VolumesN.col(i).segment(Volumes.rows() * 2 + 1, Volumes.rows() * 3) = Volumes;
316  }
317  }
318 
319  Ar = Eig_Modes.transpose() * A * Eig_Modes;
320  br = Eig_Modes.transpose() * b;
321  std::tuple <Eigen::MatrixXd, Eigen::VectorXd> tupla;
322  tupla = std::make_tuple(Ar, br);
323  return tupla;
324 }
