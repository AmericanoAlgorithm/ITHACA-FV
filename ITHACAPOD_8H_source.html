324 template<typename type_matrix>
325 std::tuple<List<Eigen::SparseMatrix<double>>, List<Eigen::VectorXd>>
326  ITHACAPOD::DEIMmodes(PtrList<type_matrix>& MatrixList, int nmodesA, int nmodesB,
327  word MatrixName)
328 {
329  List<Eigen::SparseMatrix<double>> ModesA(nmodesA);
330  List<Eigen::VectorXd> ModesB(nmodesB);
331 
332  if (!ITHACAutilities::check_folder("./ITHACAoutput/DEIM/" + MatrixName))
333  {
334  M_Assert(nmodesA <= MatrixList.size() - 2
335  && nmodesB <= MatrixList.size() - 2,
336  "The number of requested modes cannot be bigger than the number of Snapshots - 2");
337  std::tuple<List<Eigen::SparseMatrix<double>>, List<Eigen::VectorXd>> snapshots =
338  Foam2Eigen::LFvMatrix2LSM(MatrixList);
339  Eigen::MatrixXd corMatrixA = ITHACAPOD::corMatrix(std::get<0>(snapshots));
340  Eigen::MatrixXd corMatrixB = ITHACAPOD::corMatrix(std::get<1>(snapshots));
341  Info << "####### Performing the POD decomposition for the Matrix List #######"
342  << endl;
343  Spectra::DenseSymMatProd<double> opA(corMatrixA);
344  Spectra::DenseSymMatProd<double> opB(corMatrixB);
345  int ncvA = MatrixList.size();
346  int ncvB = MatrixList.size();
347  Spectra::SymEigsSolver< double, Spectra::LARGEST_ALGE, Spectra:: DenseSymMatProd<double>>
348  esA(&opA, nmodesA, ncvA);
349  Spectra::SymEigsSolver< double, Spectra::LARGEST_ALGE, Spectra:: DenseSymMatProd<double>>
350  esB(&opB, nmodesB, ncvB);
351  esA.init();
352  esB.init();
353  esA.compute(1000, 1e-10, Spectra::LARGEST_ALGE);
354  M_Assert(esA.info() == Spectra::SUCCESSFUL,
355  "The Eigenvalue Decomposition did not succeed");
356 
357  if (nmodesB != 1)
358  {
359  esB.compute(1000, 1e-10, Spectra::LARGEST_ALGE);
360  M_Assert(esB.info() == Spectra::SUCCESSFUL,
361  "The Eigenvalue Decomposition did not succeed");
362  }
363 
364  Info << "####### End of the POD decomposition for the Matrix List #######" <<
365  endl;
366  Eigen::VectorXd eigenValueseigA;
367  Eigen::MatrixXd eigenVectorseigA;
368  Eigen::VectorXd eigenValueseigB;
369  Eigen::MatrixXd eigenVectorseigB;
370  eigenValueseigA = esA.eigenvalues().real();
371  eigenVectorseigA = esA.eigenvectors().real();
372 
373  if (nmodesB != 1)
374  {
375  eigenValueseigB = esB.eigenvalues().real();
376  eigenVectorseigB = esB.eigenvectors().real();
377  }
378  else
379  {
380  eigenValueseigB.resize(1);
381  eigenVectorseigB.resize(MatrixList.size(), nmodesB);
382  eigenValueseigB(0) = 1;
383  eigenVectorseigB = eigenVectorseigB * 0;
384  eigenVectorseigB(0, 0) = 1;
385  }
386 
387  Eigen::SparseMatrix<double> tmp_A;
388  Eigen::VectorXd tmp_B;
389 
390  for (label i = 0; i < nmodesA; i++)
391  {
392  tmp_A = eigenVectorseigA(0, i) * std::get<0>(snapshots)[0];
393 
394  for (label k = 1; k < MatrixList.size(); k++)
395  {
396  tmp_A += eigenVectorseigA(k, i) * std::get<0>(snapshots)[k];
397  }
398 
399  ModesA[i] = tmp_A;
400  }
401 
402  for (label i = 0; i < nmodesB; i++)
403  {
404  tmp_B = eigenVectorseigB(0, i) * std::get<1>(snapshots)[0];
405 
406  for (label k = 1; k < MatrixList.size(); k++)
407  {
408  tmp_B += eigenVectorseigB(k, i) * std::get<1>(snapshots)[k];
409  }
410 
411  ModesB[i] = tmp_B;
412  }
413 
414  eigenValueseigA = eigenValueseigA / eigenValueseigA.sum();
415  eigenValueseigB = eigenValueseigB / eigenValueseigB.sum();
416  Eigen::VectorXd cumEigenValuesA(eigenValueseigA);
417  Eigen::VectorXd cumEigenValuesB(eigenValueseigB);
418 
419  for (int j = 1; j < cumEigenValuesA.size(); ++j)
420  {
421  cumEigenValuesA(j) += cumEigenValuesA(j - 1);
422  }
423 
424  for (int j = 1; j < cumEigenValuesB.size(); ++j)
425  {
426  cumEigenValuesB(j) += cumEigenValuesB(j - 1);
427  }
428 
429  for (int i = 0; i < ModesA.size(); i++)
430  {
431  ITHACAstream::SaveSparseMatrix(ModesA[i],
432  "./ITHACAoutput/DEIM/" + MatrixName + "/", "A_" + MatrixName + name(i));
433  }
434 
435  for (int i = 0; i < ModesB.size(); i++)
436  {
437  ITHACAstream::SaveDenseMatrix(ModesB[i],
438  "./ITHACAoutput/DEIM/" + MatrixName + "/", "B_" + MatrixName + name(i));
439  }
440 
441  ITHACAparameters para;
442  Eigen::saveMarketVector(eigenValueseigA,
443  "./ITHACAoutput/DEIM/" + MatrixName + "/eigenValuesA", para.precision,
444  para.outytpe);
445  Eigen::saveMarketVector(eigenValueseigB,
446  "./ITHACAoutput/DEIM/" + MatrixName + "/eigenValuesB", para.precision,
447  para.outytpe);
448  Eigen::saveMarketVector(cumEigenValuesA,
449  "./ITHACAoutput/DEIM/" + MatrixName + "/cumEigenValuesA", para.precision,
450  para.outytpe);
451  Eigen::saveMarketVector(cumEigenValuesB,
452  "./ITHACAoutput/DEIM/" + MatrixName + "/cumEigenValuesB", para.precision,
453  para.outytpe);
454  }
455  else
456  {
457  for (label i = 0; i < nmodesA; i++)
458  {
459  ITHACAstream::ReadSparseMatrix(ModesA[i],
460  "./ITHACAoutput/DEIM/" + MatrixName + "/", "A_" + MatrixName + name(i));
461  }
462 
463  for (label i = 0; i < nmodesB; i++)
464  {
465  ITHACAstream::ReadDenseMatrix(ModesB[i],
466  "./ITHACAoutput/DEIM/" + MatrixName + "/", "B_" + MatrixName + name(i));
467  }
468  }
469 
470  std::tuple <List<Eigen::SparseMatrix<double>>, List<Eigen::VectorXd>> tupla;
471  tupla = std::make_tuple(ModesA, ModesB);
472  return tupla;
473 }

476 template<typename T>
477 PtrList<GeometricField<T, fvPatchField, volMesh>> ITHACAPOD::DEIMmodes(
478  PtrList<GeometricField<T, fvPatchField, volMesh>>& SnapShotsMatrix, int nmodes,
479  word FunctionName)
480 {
481  if (nmodes == 0)
482  {
483  nmodes = SnapShotsMatrix.size() - 2;
484  }
485 
486  M_Assert(nmodes <= SnapShotsMatrix.size() - 2,
487  "The number of requested modes cannot be bigger than the number of Snapshots - 2");
488 
489  if (nmodes > SnapShotsMatrix.size() - 2)
490  {
491  std::cout <<
492  "The number of requested modes cannot be bigger than the number of Snapshots - 2"
493  << std::endl;
494  exit(0);
495  }
496 
497  PtrList<GeometricField<T, fvPatchField, volMesh>> Bases;
498  PtrList<GeometricField<T, fvPatchField, volMesh>> modes;
499  Bases.resize(nmodes);
500  modes.resize(nmodes);
501  Eigen::MatrixXd _corMatrix;
502  scalarField eigenValues(nmodes);
503  scalarField cumEigenValues(nmodes);
504  List<scalarField> eigenVector(nmodes);
505 
506  for (label i = 0; i < nmodes; i++)
507  {
508  eigenVector[i].setSize(SnapShotsMatrix.size());
509  }
510 
511  _corMatrix = ITHACAPOD::corMatrix(SnapShotsMatrix);
512  Info << "####### Performing the POD decomposition for " <<
513  SnapShotsMatrix[0].name() << " #######" << endl;
514  int ncv = SnapShotsMatrix.size();
515  Spectra::DenseGenMatProd<double> op(_corMatrix);
516  Spectra::GenEigsSolver< double, Spectra::LARGEST_REAL, Spectra::DenseGenMatProd<double>>
517  es(&op, nmodes, ncv);
518  es.init();
519  es.compute(1000, 1e-10, Spectra::LARGEST_REAL);
520  Info << "####### End of the POD decomposition for " << SnapShotsMatrix[0].name()
521  << " #######" << endl;
522  Eigen::VectorXd eigenValueseig;
523  Eigen::MatrixXd eigenVectoreig;
524 
525  if (es.info() == Spectra::SUCCESSFUL)
526  {
527  eigenValueseig = es.eigenvalues().real();
528  eigenVectoreig = es.eigenvectors().real();
529  }
530 
531  cumEigenValues[0] = eigenValueseig(0) / eigenValueseig.sum();
532  eigenValues[0] = eigenValueseig(0) / eigenValueseig.sum();
533 
534  for (label i = 1; i < nmodes; i++)
535  {
536  cumEigenValues[i] = cumEigenValues[i - 1] + eigenValueseig(
537  i) / eigenValueseig.sum();
538  eigenValues[i] = eigenValueseig(i) / eigenValueseig.sum();
539  }
540 
541  for (label i = 0; i < nmodes; i++)
542  {
543  for (label k = 0; k < SnapShotsMatrix.size(); k++)
544  {
545  eigenVector[i][k] = eigenVectoreig(k, i);
546  }
547  }
548 
549  auto tmb_bu = SnapShotsMatrix[0];
550  tmb_bu.rename(SnapShotsMatrix[0].name());
551 
552  for (label i = 0; i < nmodes; i++)
553  {
554  tmb_bu = eigenVector[i][0] * SnapShotsMatrix[0];
555 
556  for (label k = 1; k < SnapShotsMatrix.size(); k++)
557  {
558  tmb_bu += eigenVector[i][k] * SnapShotsMatrix[k];
559  }
560 
561  Bases.set(i, tmb_bu);
562  Info << "creating the bases " << i << " for " << SnapShotsMatrix[0].name() <<
563  endl;
564  }
565 
566  ITHACAPOD::normalizeBases(Bases);
567 
568  for (label i = 0; i < Bases.size(); i++)
569  {
570  auto tmp(Bases[i]);
571  tmp.rename(SnapShotsMatrix[0].name());
572  modes.set(i, tmp);
573  }
574 
575  Info << "####### Saving the POD bases for " << SnapShotsMatrix[0].name() <<
576  " #######" << endl;
577  ITHACAutilities::createSymLink("./ITHACAoutput/DEIM");
578 
579  for (int i = 0; i < modes.size(); i++)
580  {
581  ITHACAstream::exportSolution(modes[i], name(i + 1), "./ITHACAoutput/DEIM",
582  modes[i].name());
583  }
584 
585  ITHACAstream::exportList(eigenValues, "./ITHACAoutput/DEIM/",
586  "eigenValues_" + SnapShotsMatrix[0].name());
587  ITHACAstream::exportList(cumEigenValues, "./ITHACAoutput/DEIM/",
588  "cumEigenValues_" + SnapShotsMatrix[0].name());
589  return modes;
590 }
