48 class DEIM_function : public DEIM<PtrList<fvScalarMatrix>, volScalarField >
49 {
50 public:
51  using DEIM::DEIM;
52  static fvScalarMatrix evaluate_expression(volScalarField& T, Eigen::MatrixXd mu)
53  {
54  volScalarField yPos = T.mesh().C().component(vector::Y);
55  volScalarField xPos = T.mesh().C().component(vector::X);
56  volScalarField nu(T);
57  for (auto i = 0; i < nu.size(); i++)
58  {
59  nu[i] = std::exp( - 2 * std::pow(xPos[i] - mu(0) - 1, 2) - 2 * std::pow(yPos[i] - mu(1) - 0.5 , 2)) + 1;
60  }
61  nu.correctBoundaryConditions();
62  dimensionedScalar correct
63  (
64  "correct",
65  dimensionSet(0, 1, 0, 0, 0, 0, 0),
66  scalar(3.0)
67  );
68 
69  fvScalarMatrix TiEqn22
70  (
71  fvm::laplacian(nu, T, "Gauss linear")
72  );
73  return TiEqn22;
74  }
75 
76  Eigen::MatrixXd onlineCoeffsA(Eigen::MatrixXd mu)
77  {
78  Eigen::MatrixXd theta(fieldsA.size(), 1);
79  for (int i = 0; i < fieldsA.size(); i++)
80  {
81  Eigen::SparseMatrix<double> Mr;
82  Eigen::VectorXd br;
83  fvScalarMatrix Aof = evaluate_expression(fieldsA[i], mu);
84  Foam2Eigen::fvMatrix2Eigen(Aof, Mr, br);
85  int ind_row = localMagicPointsA[i].first() + xyz_A[i].first() * fieldsA[i].size();
86  int ind_col = localMagicPointsA[i].second() + xyz_A[i].second() * fieldsA[i].size();
87  theta(i) = Mr.coeffRef(ind_row, ind_col);
88  }
89  return theta;
90  }
91 
92  Eigen::MatrixXd onlineCoeffsB(Eigen::MatrixXd mu)
93  {
94  Eigen::MatrixXd theta(fieldsB.size(), 1);
95  for (int i = 0; i < fieldsB.size(); i++)
96  {
97  Eigen::SparseMatrix<double> Mr;
98  Eigen::VectorXd br;
99  fvScalarMatrix Aof = evaluate_expression(fieldsB[i], mu);
100  Foam2Eigen::fvMatrix2Eigen(Aof, Mr, br);
101  int ind_row = localMagicPointsB[i] + xyz_B[i] * fieldsB[i].size();
102  theta(i) = br(ind_row);
103  }
104  return theta;
105  }

196  void PODDEIM(int NmodesT, int NmodesDEIMA, int NmodesDEIMB)
197  {
198  DEIMmatrice = new DEIM_function(Mlist, NmodesDEIMA, NmodesDEIMB, "T_matrix");
199  fvMesh& mesh = const_cast<fvMesh&>(T.mesh());
200  DEIMmatrice->generateSubmeshesMatrix(2, mesh, T);
201  DEIMmatrice->generateSubmeshesVector(2, mesh, T);
202  ModesTEig = Foam2Eigen::PtrList2Eigen(Tmodes);
203  ModesTEig.conservativeResize(ModesTEig.rows(), NmodesT);
204  ReducedMatricesA.resize(NmodesDEIMA);
205  ReducedVectorsB.resize(NmodesDEIMB);
206  for (int i = 0; i < NmodesDEIMA; i++)
207  {
208  ReducedMatricesA[i] = ModesTEig.transpose() * DEIMmatrice->MatrixOnlineA[i] * ModesTEig;
209  }
210  for (int i = 0; i < NmodesDEIMB; i++)
211  {
212  ReducedVectorsB[i] = ModesTEig.transpose() * DEIMmatrice->MatrixOnlineB;
213  }
214  };

216  void OnlineSolve(Eigen::MatrixXd par_new, word Folder)
217  {
218  auto t1 = std::chrono::high_resolution_clock::now();
219  auto t2 = std::chrono::high_resolution_clock::now();
220  auto time_span = std::chrono::duration_cast<std::chrono::duration<double>>(t2 - t1);
221  time_rom = 0;
222 
223  for (int i = 0; i < par_new.rows(); i++)
224  {
225  // solve
226  t1 = std::chrono::high_resolution_clock::now();
227  Eigen::MatrixXd thetaonA = DEIMmatrice->onlineCoeffsA(par_new.row(i));
228  Eigen::MatrixXd A = EigenFunctions::MVproduct(ReducedMatricesA, thetaonA);
229  Eigen::VectorXd B = ReducedVectorsB[0];
230  Eigen::VectorXd x = A.ldlt().solve(B);
231  Eigen::VectorXd full = ModesTEig * x;
232  t2 = std::chrono::high_resolution_clock::now();
233  time_span = std::chrono::duration_cast<std::chrono::duration<double>>(t2 - t1);
234  time_rom += time_span.count();
235  // Export
236  volScalarField Tred("Tred", T);
237  Tred = Foam2Eigen::Eigen2field(Tred, full);
238  ITHACAutilities::exportSolution(Tred, "./ITHACAoutput/" + Folder , name(i + 1));
239  Tonline.append(Tred);
240  }
241  }

244 int main(int argc, char *argv[])
245 {
246  // Construct the case
247  DEIMlaplacian example(argc, argv);
248 
249  // Create the offline parameters for the solve
250  example.mu = ITHACAutilities::rand(100, 2, -0.5, 0.5);
251 
252  // Solve the offline problem to compute the snapshots for the projections
253  example.OfflineSolve(example.mu, "Offline");
254 
255  // Compute the POD modes
256  ITHACAPOD::getModes(example.Tfield, example.Tmodes, example.podex, 0, 0, 20);
257 
258  // Compute the offline part of the DEIM procedure
259  example.PODDEIM();
260 
261  // Construct a new set of parameters
262  Eigen::MatrixXd par_new1 = ITHACAutilities::rand(100, 2, -0.5, 0.5);
263 
264  // Solve the online problem with the new parameters
265  example.OnlineSolve(par_new1, "Online_red");
266 
267  // Solve a new full problem with the new parameters (necessary to compute speed up and error)
268  DEIMlaplacian example_new(argc, argv);
269  example_new.OnlineSolveFull(par_new1, "Online_full");
270 
271  // Output some infos
272  std::cout << std::endl << "The FOM Solve took: " << example_new.time_full << " seconds." << std::endl;
273  std::cout << std::endl << "The ROM Solve took: " << example.time_rom << " seconds." << std::endl;
274  std::cout << std::endl << "The Speed-up is: " << example_new.time_full / example.time_rom << std::endl << std::endl;
275 
276  Eigen::MatrixXd error = ITHACAutilities::error_listfields(example_new.Tfield, example.Tonline);
277 
278  std::cout << "The mean L2 error is: " << error.mean() << std::endl;
279 
280  exit(0);
281 }
