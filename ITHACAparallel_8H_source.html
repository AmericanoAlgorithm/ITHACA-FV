81 template<class type_of_field>
82 ITHACAparallel<type_of_field>::ITHACAparallel(
83  GeometricField<type_of_field, fvPatchField, volMesh>& field)
84 {
85  // IF construction
86  indices = new labelIOList(
87  IOobject
88  (
89  "cellProcAddressing",
90  field.mesh().facesInstance(),
91  field.mesh().meshSubDir,
92  field.mesh(),
93  IOobject::MUST_READ,
94  IOobject::NO_WRITE
95  )
96  );
97  N_IF_glob = (field.size());
98  reduce(N_IF_glob, sumOp<int>());
99  // BF construction
100  startFaces = new List<labelList> (field.boundaryFieldRef().size(),
101  labelList(Pstream::nProcs(), 0));
102  Gsize_BF = new List<labelList> (field.boundaryFieldRef().size(),
103  labelList(Pstream::nProcs(), 0));
104 
105  for (int i = 0; i < field.boundaryFieldRef().size(); i++)
106  {
107  Gsize_BF()[i][Pstream::myProcNo()] = field.boundaryFieldRef()[i].size();
108  reduce(Gsize_BF()[i], sumOp<labelList>());
109  }
110 
111  for (int i = 0; i < field.boundaryFieldRef().size(); i++)
112  {
113  for (label k = 1; k < Pstream::nProcs(); k++)
114  {
115  startFaces()[i][k] = startFaces()[i][k - 1] + Gsize_BF()[i][k - 1];
116  }
117  }
